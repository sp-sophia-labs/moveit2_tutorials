507 void MotionPlanningDisplay::computeMetricsInternal(std::map<std::string, double>& metrics,
508  const robot_interaction::EndEffectorInteraction& ee,
509  const moveit::core::RobotState& state, double payload)
510 {
511  metrics.clear();
512  dynamics_solver::DynamicsSolverPtr ds;
513  std::map<std::string, dynamics_solver::DynamicsSolverPtr>::const_iterator it = dynamics_solver_.find(ee.parent_group);
514  if (it != dynamics_solver_.end())
515  ds = it->second;
516 
517  // Max payload
518  if (ds)
519  {
520  double max_payload;
521  unsigned int saturated_joint;
522  std::vector<double> joint_values;
523  state.copyJointGroupPositions(ee.parent_group, joint_values);
524  if (ds->getMaxPayload(joint_values, max_payload, saturated_joint))
525  {
526  metrics["max_payload"] = max_payload;
527  metrics["saturated_joint"] = saturated_joint;
528  }
529  std::vector<double> joint_torques;
530  joint_torques.resize(joint_values.size());
531  if (ds->getPayloadTorques(joint_values, payload, joint_torques))
532  {
533  for (std::size_t i = 0; i < joint_torques.size(); ++i)
534  {
535  std::stringstream stream;
536  stream << "torque[" << i << "]";
537  metrics[stream.str()] = joint_torques[i];
538  }
539  }
540  }
541 
542  if (kinematics_metrics_)
543  {
544  if (position_only_ik_.find(ee.parent_group) == position_only_ik_.end())
545  node_->get_parameter_or(ee.parent_group + ".position_only_ik", position_only_ik_[ee.parent_group], false);
546 
547  double manipulability_index, manipulability;
548  bool position_ik = position_only_ik_[ee.parent_group];
549  if (kinematics_metrics_->getManipulabilityIndex(state, ee.parent_group, manipulability_index, position_ik))
550  metrics["manipulability_index"] = manipulability_index;
551  if (kinematics_metrics_->getManipulability(state, ee.parent_group, manipulability))
552  metrics["manipulability"] = manipulability;
553  }
554 }

567 void MotionPlanningDisplay::displayMetrics(bool start)
568 {
569  if (!robot_interaction_ || !planning_scene_monitor_)
570  return;
571 
572  static const Ogre::Quaternion ORIENTATION(1.0, 0.0, 0.0, 0.0);
573  const std::vector<robot_interaction::EndEffectorInteraction>& eef = robot_interaction_->getActiveEndEffectors();
574  if (eef.empty())
575  return;
576 
577  moveit::core::RobotStateConstPtr state = start ? getQueryStartState() : getQueryGoalState();
578 
579  for (const robot_interaction::EndEffectorInteraction& ee : eef)
580  {
581  Ogre::Vector3 position(0.0, 0.0, 0.0);
582  std::map<std::string, double> text_table;
583  const std::map<std::string, double>& metrics_table = computed_metrics_[std::make_pair(start, ee.parent_group)];
584  if (compute_weight_limit_property_->getBool())
585  {
586  copyItemIfExists(metrics_table, text_table, "max_payload");
587  copyItemIfExists(metrics_table, text_table, "saturated_joint");
588  }
589  if (show_manipulability_index_property_->getBool())
590  copyItemIfExists(metrics_table, text_table, "manipulability_index");
591  if (show_manipulability_property_->getBool())
592  copyItemIfExists(metrics_table, text_table, "manipulability");
593  if (show_joint_torques_property_->getBool())
594  {
595  std::size_t nj = getRobotModel()->getJointModelGroup(ee.parent_group)->getJointModelNames().size();
596  for (size_t j = 0; j < nj; ++j)
597  {
