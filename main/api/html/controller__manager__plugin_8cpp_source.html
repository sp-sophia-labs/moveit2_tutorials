132  void discover(bool force = false)
133  {
134  // Skip if controller stamp is too new for new discovery, enforce update if force==true
135  if (!force && ((node_->now() - controllers_stamp_) < CONTROLLER_INFORMATION_VALIDITY_AGE))
136  return;
137 
138  controllers_stamp_ = node_->now();
139 
140  auto request = std::make_shared<controller_manager_msgs::srv::ListControllers::Request>();
141  auto result_future = list_controllers_service_->async_send_request(request);
142  if (result_future.wait_for(std::chrono::duration<double>(SERVICE_CALL_TIMEOUT)) == std::future_status::timeout)
143  {
144  RCLCPP_WARN_STREAM(LOGGER, "Failed to read controllers from " << list_controllers_service_->get_service_name()
145  << " within " << SERVICE_CALL_TIMEOUT
146  << " seconds");
147  return;
148  }
149 
150  managed_controllers_.clear();
151  active_controllers_.clear();
152 
153  auto result = result_future.get();
154  if (!Ros2ControlManager::fixChainedControllers(result))
155  {
156  return;
157  }
158 
159  for (const controller_manager_msgs::msg::ControllerState& controller : result->controller)
160  {
161  // If the controller is active, add it to the map of active controllers.
162  if (isActive(controller))
163  {
164  // Get the names of the interfaces currently claimed by the active controller.
165  auto& claimed_interfaces = active_controllers_.insert(std::make_pair(controller.name, controller))
166  .first->second.claimed_interfaces; // without namespace
167  // Modify the claimed interface names in-place to only include the name of the joint and not the command type
168  // (e.g. position, velocity, etc.).
169  std::transform(claimed_interfaces.cbegin(), claimed_interfaces.cend(), claimed_interfaces.begin(),
170  [](const std::string& claimed_interface) {
171  return parseJointNameFromResource(claimed_interface);
172  });
173  }
174 
175  // Instantiate a controller handle if one is available for this type of controller.
176  if (loader_.isClassAvailable(controller.type))
177  {
178  std::string absname = getAbsName(controller.name);
179  auto controller_it = managed_controllers_.insert(std::make_pair(absname, controller)).first; // with namespace
180  // Get the names of the interfaces that would be claimed by this currently-inactive controller if it was activated.
181  auto& required_interfaces = controller_it->second.required_command_interfaces;
182  // Modify the required interface names in-place to only include the name of the joint and not the command type
183  // (e.g. position, velocity, etc.).
184  std::transform(required_interfaces.cbegin(), required_interfaces.cend(), required_interfaces.begin(),
185  [](const std::string& required_interface) {
186  return parseJointNameFromResource(required_interface);
187  });
188  allocate(absname, controller_it->second);
189  }
190  }
191  }

199  void allocate(const std::string& name, const controller_manager_msgs::msg::ControllerState& controller)
200  {
201  if (handles_.find(name) == handles_.end())
202  {
203  const std::string& type = controller.type;
204  AllocatorsMap::iterator alloc_it = allocators_.find(type);
205  if (alloc_it == allocators_.end())
206  { // create allocator is needed
207  alloc_it = allocators_.insert(std::make_pair(type, loader_.createUniqueInstance(type))).first;
208  }
209 
210  std::vector<std::string> resources;
211  // Collect claimed resources across different hardware interfaces
212  for (const auto& resource : controller.claimed_interfaces)
213  {
214  resources.push_back(parseJointNameFromResource(resource));
215  }
216 
217  moveit_controller_manager::MoveItControllerHandlePtr handle =
218  alloc_it->second->alloc(node_, name, resources); // allocate handle
219  if (handle)
220  handles_.insert(std::make_pair(name, handle));
221  }
222  }

362  bool switchControllers(const std::vector<std::string>& activate_base,
363  const std::vector<std::string>& deactivate_base) override
364  {
365  // add controller dependencies
366  std::vector<std::string> activate = activate_base;
367  std::vector<std::string> deactivate = deactivate_base;
368  for (auto controllers : { &activate, &deactivate })
369  {
370  auto queue = *controllers;
371  while (!queue.empty())
372  {
373  auto controller = queue.back();
374  controller.erase(0, 1);
375  queue.pop_back();
376  for (const auto& dependency : dependency_map_[controller])
377  {
378  queue.push_back(dependency);
379  controllers->push_back("/" + dependency);
380  }
381  }
382  }
383  // activation dependencies must be started first, but they are processed last, so the order needs to be flipped
384  std::reverse(activate.begin(), activate.end());
385 
386  std::scoped_lock<std::mutex> lock(controllers_mutex_);
387  discover(true);
388 
389  typedef boost::bimap<std::string, std::string> resources_bimap;
390 
391  resources_bimap claimed_resources;
392 
393  // fill bimap with active controllers and their resources
394  for (std::pair<const std::string, controller_manager_msgs::msg::ControllerState>& active_controller :
395  active_controllers_)
396  {
397  for (const auto& resource : active_controller.second.claimed_interfaces)
398  {
399  claimed_resources.insert(resources_bimap::value_type(active_controller.second.name, resource));
400  }
401  }
402 
403  auto request = std::make_shared<controller_manager_msgs::srv::SwitchController::Request>();
404  for (const std::string& it : deactivate)
405  {
406  ControllersMap::iterator c = managed_controllers_.find(it);
407  if (c != managed_controllers_.end())
408  { // controller belongs to this manager
409  request->stop_controllers.push_back(c->second.name);
410  claimed_resources.right.erase(c->second.name); // remove resources
411  }
412  }
413 
414  // For each controller to activate, find conflicts between the interfaces required by that controller and the
415  // interfaces claimed by currently-active controllers.
416  for (const std::string& it : activate)
417  {
418  ControllersMap::iterator c = managed_controllers_.find(it);
419  if (c != managed_controllers_.end())
420  { // controller belongs to this manager
421  request->start_controllers.push_back(c->second.name);
422  for (const auto& required_resource : c->second.required_command_interfaces)
423  {
424  resources_bimap::right_const_iterator res = claimed_resources.right.find(required_resource);
425  if (res != claimed_resources.right.end())
426  { // resource is claimed
427  if (std::find(request->stop_controllers.begin(), request->stop_controllers.end(), res->second) ==
428  request->stop_controllers.end())
429  {
430  request->stop_controllers.push_back(res->second); // add claiming controller to stop list
431  }
432  claimed_resources.left.erase(res->second); // remove claimed resources
433  }
434  }
435  }
436  }
437 
438  // Setting level to STRICT means that the controller switch will only be committed if all controllers are
439  // successfully activated or deactivated.
440  request->strictness = controller_manager_msgs::srv::SwitchController::Request::STRICT;
441 
442  if (!request->start_controllers.empty() || !request->stop_controllers.empty())
443  { // something to switch?
444  auto result_future = switch_controller_service_->async_send_request(request);
445  if (result_future.wait_for(std::chrono::duration<double>(SERVICE_CALL_TIMEOUT)) == std::future_status::timeout)
446  {
447  RCLCPP_ERROR_STREAM(LOGGER, "Couldn't switch controllers at " << switch_controller_service_->get_service_name()
448  << " within " << SERVICE_CALL_TIMEOUT
449  << " seconds");
450  return false;
451  }
452  discover(true);
453  return result_future.get()->ok;
454  }
455  return true;
456  }
462  bool fixChainedControllers(std::shared_ptr<controller_manager_msgs::srv::ListControllers::Response>& result)
463  {
464  std::unordered_map<std::string, size_t> controller_name_map;
465  for (size_t i = 0; i < result->controller.size(); ++i)
466  {
467  controller_name_map[result->controller[i].name] = i;
468  }
469  for (auto& controller : result->controller)
470  {
471  if (controller.chain_connections.size() > 1)
472  {
473  RCLCPP_ERROR_STREAM(LOGGER, "Controller with name %s chains to more than one controller. Chaining to more than "
474  "one controller is not supported.");
475  return false;
476  }
477  dependency_map_[controller.name].clear();
478  for (const auto& chained_controller : controller.chain_connections)
479  {
480  auto ind = controller_name_map[chained_controller.name];
481  dependency_map_[controller.name].push_back(chained_controller.name);
482  controller.required_command_interfaces = result->controller[ind].required_command_interfaces;
483  controller.claimed_interfaces = result->controller[ind].claimed_interfaces;
484  result->controller[ind].claimed_interfaces.clear();
485  result->controller[ind].required_command_interfaces.clear();
486  }
487  }
488 
489  return true;
490  }
