91  RCLCPP_DEBUG(LOGGER,
92  "Choosing tip frame of kinematic solver for group %s"

114  kinematics::KinematicsBasePtr allocKinematicsSolver(const moveit::core::JointModelGroup* jmg)
115  {
116  kinematics::KinematicsBasePtr result;
117  if (!kinematics_loader_)
118  {
119  RCLCPP_ERROR(LOGGER, "Invalid kinematics loader.");
120  return result;
121  }
122  if (!jmg)
123  {
124  RCLCPP_ERROR(LOGGER, "Specified group is nullptr. Cannot allocate kinematics solver.");
125  return result;
126  }
127  const std::vector<const moveit::core::LinkModel*>& links = jmg->getLinkModels();
128  if (links.empty())
129  {
130  RCLCPP_ERROR(LOGGER, "No links specified for group '%s'. Cannot allocate kinematics solver.",
131  jmg->getName().c_str());
132  return result;
133  }
134 
135  RCLCPP_DEBUG(LOGGER, "Trying to allocate kinematics solver for group '%s'", jmg->getName().c_str());
136 
137  const std::string& base = links.front()->getParentJointModel()->getParentLinkModel() ?
138  links.front()->getParentJointModel()->getParentLinkModel()->getName() :
139  jmg->getParentModel().getModelFrame();
140 
141  // just to be sure, do not call the same pluginlib instance allocation function in parallel
142  std::scoped_lock slock(lock_);
143  for (auto const& [group, solver] : possible_kinematics_solvers_)
144  {
145  try
146  {
147  result = kinematics_loader_->createUniqueInstance(solver);
148  if (result)
149  {
150  // choose the tip of the IK solver
151  const std::vector<std::string> tips = chooseTipFrames(jmg);
152 
153  // choose search resolution
154  double search_res = search_res_.find(jmg->getName())->second; // we know this exists, by construction
155 
156  if (!result->initialize(node_, jmg->getParentModel(), jmg->getName(),
157  (base.empty() || base[0] != '/') ? base : base.substr(1), tips, search_res))
158  {
159  RCLCPP_ERROR(LOGGER, "Kinematics solver of type '%s' could not be initialized for group '%s'",
160  solver.c_str(), jmg->getName().c_str());
161  result.reset();
162  continue;
163  }
164 
165  result->setDefaultTimeout(jmg->getDefaultIKTimeout());
166  RCLCPP_DEBUG(LOGGER,
167  "Successfully allocated and initialized a kinematics solver of type '%s' with search "
168  "resolution %lf for group '%s' at address %p",
169  solver.c_str(), search_res, jmg->getName().c_str(), result.get());
170  break;
171  }
172  }
173  catch (pluginlib::PluginlibException& e)
174  {
175  RCLCPP_ERROR(LOGGER, "The kinematics plugin (%s) failed to load. Error: %s", solver.c_str(), e.what());
176  }
177  }
178 
179  if (!result)
180  {
181  RCLCPP_DEBUG(LOGGER, "No usable kinematics solver was found for this group.\n"
182  "Did you load kinematics.yaml into your node's namespace?");
183  }
184  return result;
185  }
186 
187  // cache solver between two consecutive calls
188  // first call in RobotModelLoader::loadKinematicsSolvers() is just to check suitability for jmg
189  // second call in JointModelGroup::setSolverAllocators() is to actually retrieve the instance for use
190  kinematics::KinematicsBasePtr allocKinematicsSolverWithCache(const moveit::core::JointModelGroup* jmg)
191  {
192  std::scoped_lock slock(cache_lock_);
193  kinematics::KinematicsBasePtr& cached = instances_[jmg];
194  if (cached.unique())
195  return std::move(cached); // pass on unique instance
196 
197  // create a new instance and store in instances_
198  cached = allocKinematicsSolver(jmg);
199  return cached;
200  }

234 moveit::core::SolverAllocatorFn KinematicsPluginLoader::getLoaderFunction(const srdf::ModelSharedPtr& srdf_model)
235 {
236  if (!loader_)
237  {
238  RCLCPP_DEBUG(LOGGER, "Configuring kinematics solvers");
239  groups_.clear();
240 
241  std::map<std::string, std::string> possible_kinematics_solvers;
242  std::map<std::string, double> search_res;
243  std::map<std::string, std::vector<std::string>> iksolver_to_tip_links;
244 
245  if (srdf_model)
246  {
247  const std::vector<srdf::Model::Group>& known_groups = srdf_model->getGroups();
248 
249  RCLCPP_DEBUG(LOGGER, "Loading kinematics parameters...");
250  // read the list of plugin names for possible kinematics solvers
251  for (const srdf::Model::Group& known_group : known_groups)
252  {
253  groups_.push_back(known_group.name_);
254 
255  std::string kinematics_param_prefix = robot_description_ + "_kinematics." + known_group.name_;
256  group_param_listener_.try_emplace(known_group.name_,
257  std::make_shared<kinematics::ParamListener>(node_, kinematics_param_prefix));
258  group_params_.try_emplace(known_group.name_, group_param_listener_.at(known_group.name_)->get_params());
259 
260  std::string kinematics_solver_param_name = kinematics_param_prefix + ".kinematics_solver";
261  const auto kinematics_solver = group_params_.at(known_group.name_).kinematics_solver;
262  possible_kinematics_solvers[known_group.name_] = kinematics_solver;
263  RCLCPP_DEBUG(LOGGER, "Found kinematics solver '%s' for group '%s'.", kinematics_solver.c_str(),
264  known_group.name_.c_str());
265 
266  std::string kinematics_solver_res_param_name = kinematics_param_prefix + ".kinematics_solver_search_resolution";
267  const auto kinematics_solver_search_resolution =
268  group_params_.at(known_group.name_).kinematics_solver_search_resolution;
269  search_res[known_group.name_] = kinematics_solver_search_resolution;
270  RCLCPP_DEBUG(LOGGER, "Found param %s : %f", kinematics_solver_res_param_name.c_str(),
271  kinematics_solver_search_resolution);
272 
273  std::string kinematics_solver_timeout_param_name = kinematics_param_prefix + ".kinematics_solver_timeout";
274  const auto kinematics_solver_timeout = group_params_.at(known_group.name_).kinematics_solver_timeout;
275  ik_timeout_[known_group.name_] = kinematics_solver_timeout;
276  RCLCPP_DEBUG(LOGGER, "Found param %s : %f", kinematics_solver_timeout_param_name.c_str(),
277  kinematics_solver_timeout);
278  }
279  }
280 
281  loader_ =
282  std::make_shared<KinematicsLoaderImpl>(node_, robot_description_, possible_kinematics_solvers, search_res);
283  }
284 
285  return [&loader = *loader_](const moveit::core::JointModelGroup* jmg) {
286  return loader.allocKinematicsSolverWithCache(jmg);
287  };
288 }
