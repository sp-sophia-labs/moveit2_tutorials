165 // ******************************************************************************************
166 // Highlight a robot group
167 // ******************************************************************************************
168 void RVizPanel::highlightGroupEvent(const std::string& group_name)
169 {
170  auto rm = getRobotModel();
171  if (!rm)
172  return;
173  // Highlight the selected planning group by looping through the links
174  if (!rm->hasJointModelGroup(group_name))
175  return;
176 
177  const moveit::core::JointModelGroup* joint_model_group = rm->getJointModelGroup(group_name);
178  if (joint_model_group)
179  {
180  const std::vector<const moveit::core::LinkModel*>& link_models = joint_model_group->getLinkModels();
181  // Iterate through the links
182  for (std::vector<const moveit::core::LinkModel*>::const_iterator link_it = link_models.begin();
183  link_it < link_models.end(); ++link_it)
184  highlightLink((*link_it)->getName(), QColor(255, 0, 0));
185  }
186 }

188 // ******************************************************************************************
189 // Unhighlight all robot links
190 // ******************************************************************************************
191 void RVizPanel::unhighlightAllEvent()
192 {
193  auto rm = getRobotModel();
194  if (!rm)
195  return;
196  // Get the names of the all links robot
197  const std::vector<std::string>& links = rm->getLinkModelNamesWithCollisionGeometry();
198 
199  // Quit if no links found
200  if (links.empty())
201  {
202  return;
203  }
204 
205  // check if rviz is ready
206  if (!rviz_manager_ || !robot_state_display_)
207  {
208  return;
209  }
210 
211  // Iterate through the links
212  for (std::vector<std::string>::const_iterator link_it = links.begin(); link_it < links.end(); ++link_it)
213  {
214  if ((*link_it).empty())
215  continue;
216 
217  robot_state_display_->unsetLinkColor(*link_it);
218  }
219 }
