102  FileStatus getStatus() const
103  {
104  std::filesystem::path full_path = getPath();
105  if (!std::filesystem::is_regular_file(full_path) || last_gen_time_ == GeneratedTime())
106  {
107  return FileStatus::NEW;
108  }
109  GeneratedTime mod_time = std::filesystem::last_write_time(full_path);
110  if (mod_time > last_gen_time_ + TIME_MOD_TOLERANCE || mod_time < last_gen_time_ - TIME_MOD_TOLERANCE)
111  {
112  return hasChanges() ? FileStatus::CONFLICTED : FileStatus::EXTERNALLY_MODIFIED;
113  }
114  else
115  {
116  return hasChanges() ? FileStatus::CHANGED : FileStatus::UNCHANGED;
117  }
118  }
119 
120 protected:
121  static constexpr GeneratedTime::duration TIME_MOD_TOLERANCE = std::chrono::seconds(10);
122 
123  std::filesystem::path package_path_;
124  const GeneratedTime& last_gen_time_;
125 };
