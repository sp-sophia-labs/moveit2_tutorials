44 void MoveItControllersConfig::loadPrevious(const std::filesystem::path& package_path, const YAML::Node& /*node*/)
45 {
46  changed_ = false;
47  controllers_.clear();
48 
49  // Load moveit controllers yaml file if available-----------------------------------------------
50  std::filesystem::path ros_controllers_yaml_path = package_path / MOVEIT_CONTROLLERS_YAML;
51  std::ifstream input_stream(ros_controllers_yaml_path);
52  if (!input_stream.good())
53  {
54  RCLCPP_WARN_STREAM(*logger_, "Does not exist " << ros_controllers_yaml_path);
55  return;
56  }
57 
58  // Begin parsing
59  try
60  {
61  // Used in parsing controllers
62  ControllerInfo control_setting;
63  YAML::Node controllers = YAML::Load(input_stream)["moveit_simple_controller_manager"];
64 
65  std::vector<std::string> controller_names;
66  getYamlProperty(controllers, "controller_names", controller_names);
67 
68  // Loop through all controllers
69  for (const std::string& controller_name : controller_names)
70  {
71  const YAML::Node& cnode = controllers[controller_name];
72  if (!cnode.IsDefined())
73  {
74  RCLCPP_WARN_STREAM(*logger_, "Configuration for controller " << controller_name << " does not exist! Ignoring.");
75  continue;
76  }
77 
78  if (!parseController(controller_name, cnode))
79  {
80  return;
81  }
82  }
83  }
84  catch (YAML::ParserException& e) // Catch errors
85  {
86  RCLCPP_ERROR_STREAM(*logger_, e.what());
87  }
88 }

93 bool MoveItControllersConfig::parseController(const std::string& name, const YAML::Node& controller_node)
94 {
95  // Used in parsing controllers
96  ControllerInfo control_setting;
97  control_setting.name_ = name;
98  getYamlProperty(controller_node, "type", control_setting.type_);
99  if (control_setting.type_.empty())
100  {
101  RCLCPP_ERROR_STREAM(*logger_, "Couldn't parse type for controller " << name << " in moveit_controllers.yaml");
102  return false;
103  }
104 
105  for (const std::string parameter : { "action_ns", "default" })
106  {
107  if (controller_node[parameter].IsDefined())
108  {
109  control_setting.parameters_[parameter] = controller_node[parameter].as<std::string>();
110  }
111  }
112 
113  const YAML::Node& joints_node = controller_node["joints"];
114 
115  if (joints_node.IsSequence())
116  {
117  control_setting.joints_ = joints_node.as<std::vector<std::string>>();
118  }
119  else if (joints_node.IsDefined())
120  {
121  control_setting.joints_.push_back(joints_node.as<std::string>());
122  }
123  if (control_setting.joints_.empty())
124  {
125  RCLCPP_ERROR_STREAM(*logger_, "Couldn't parse joints for controller " << name << " in moveit_controllers.yaml");
126  return false;
127  }
128  // All required fields were parsed correctly
129  controllers_.push_back(control_setting);
130  return true;
131 }

137 {

149  emitter << YAML::Newline;
150  emitter << YAML::Newline;
151 
152  emitter << YAML::Key << "moveit_simple_controller_manager" << YAML::Value;
153  emitter << YAML::BeginMap;
154  {
155  emitter << YAML::Key << "controller_names";
156  emitter << YAML::Value;
157  emitter << YAML::BeginSeq;
158  for (const ControllerInfo& ci : parent_.controllers_)
159  {
160  emitter << ci.name_;
161  }
162  emitter << YAML::EndSeq;
163  emitter << YAML::Newline;
164  emitter << YAML::Newline;
165 
166  for (const auto& controller : parent_.controllers_)
167  {
168  emitter << YAML::Key << controller.name_;
169  emitter << YAML::Value;
170  emitter << YAML::BeginMap;
171  {
172  emitter << YAML::Key << "type" << YAML::Value << controller.type_;
173  // Write joints
174  emitter << YAML::Key << "joints";
175  emitter << YAML::Value;
176  emitter << YAML::BeginSeq;
177 
178  // Iterate through the joints
179  for (const std::string& joint : controller.joints_)
180  {
181  emitter << joint;
182  }
183  emitter << YAML::EndSeq;
184 
185  for (const auto& pair : controller.parameters_)
186  {
187  emitter << YAML::Key << pair.first;
188  emitter << YAML::Value << pair.second;
189  }
190  }
191  emitter << YAML::EndMap;
192  }
193  }
194  emitter << YAML::EndMap;
195  }
196  emitter << YAML::EndMap;
197 
198  return true;
